import { Meta, Title } from '@storybook/blocks';
// Load all icons
// @see https://vitejs.dev/guide/features.html#glob-import
export const illustrations = import.meta.glob('./illustrations/*.svg', {
  eager: true,
});
// Generate Storybook Icon elements for each
// @see https://storybook.js.org/docs/react/api/doc-block-icongallery
export const generateElements = (illustrations) => {
  return Object.entries(illustrations).map(([path, mod]) => {
    const name = path.split('/').pop().split('.').shift();
    return <mod.ReactComponent key={name} />;
  });
};

<Meta title="Assets/Illustrations" />

<Title>Illustrations</Title>

{generateElements(illustrations)}
