import { Meta, Title, ColorPalette, ColorItem } from '@storybook/blocks';
import tokens from '../compiled/js/tokens.json';
export const generateColorPalette = (props) => {
	return Object.values(props).map((prop) => (
		<ColorItem
			key={prop.name}
			title={`--${prop.name}`}
			subtitle={prop.comment}
			colors={[prop.value]}
		/>
	));
};

<Meta title="Brand/Colors" />

<Title>Colors</Title>

## Background Colors

<ColorPalette>{generateColorPalette(tokens.color.background)}</ColorPalette>

## Text Colors

<ColorPalette>{generateColorPalette(tokens.color.text)}</ColorPalette>

## Status Colors

<ColorPalette>{generateColorPalette(tokens.color.status)}</ColorPalette>

## Brand Colors

<ColorPalette>{generateColorPalette(tokens.color.brand)}</ColorPalette>

## Base Colors

<ColorPalette>{generateColorPalette(tokens.color.base)}</ColorPalette>
